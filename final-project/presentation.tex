\begin{frame}
\frametitle{What we aim to do}
\begin{itemize}
\item Currently, R5 allocates a new stack frame for every function call
\item In a functional language, this is terrible, since iteration relies on
      recursive functions (i.e., tons of successive function calls).
\item This can make even simple iteration use up all of one's stack space
      in the blink of an eye.
\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}[fragile]
\frametitle{One example}
\begin{minted}{haskell}
class Equal a where
  eq :: a -> a -> Bool

data Tree a = Leaf | Node a | Branch (Tree a) a (Tree a)

instance Equal a => Equal (Tree a) where
  eq Leaf Leaf = True
  eq (Node a1) (Node a2) =
      eq a1 a2
  eq (Branch lc1 a1 rc1) (Branch lc2 a2 rc2) =
      eq lc1 lc2 && eq a1 a2 && eq rc1 rc2
  eq _ _ = False
\end{minted}
\end{frame}
